def crossover_individuals(individual1, individual2, pset):
    """Randomly select crossover point in each individual and exchange each
    subtree with the point as root between each individual.

    :param individual1: First tree participating in the crossover.
    :param individual2: Second tree participating in the crossover.
    :returns: A tuple of two trees.
    """

    if len(individual1) < 2 or len(individual2) < 2:
        # No crossover on single node tree
        return individual1, individual2

    # List all available primitive types in each individual
    types1 = defaultdict(list)
    types2 = defaultdict(list)
    if individual1.root.ret == object:
        # Not STGP optimization
        types1[object] = list(range(1, len(individual1)))
        types2[object] = list(range(1, len(individual2)))
        common_types = [object]
    else:
        for idx, node in enumerate(individual1[1:], 1):
            types1[node.ret].append(idx)
        for idx, node in enumerate(individual2[1:], 1):
            types2[node.ret].append(idx)
        common_types = set(types1.keys()).intersection(set(types2.keys()))

    if len(common_types) > 0:
        type_ = random.choice(list(common_types))

        index1 = random.choice(types1[type_])
        index2 = random.choice(types2[type_])

        slice1 = individual1.searchSubtree(index1)
        slice2 = individual2.searchSubtree(index2)
        individual1[slice1], individual2[slice2] = individual2[slice2], individual1[slice1]

    return individual1, individual2
